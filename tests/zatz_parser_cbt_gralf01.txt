*&---------------------------------------------------------------------*
*&      Form  get_moneda_gral
*& llamar después de haber parseado la sociedad. Devuelve la moneda
"& parametrizada por sociedad.
*&---------------------------------------------------------------------*
FORM get_moneda_gral USING pv_soc
                           pv_parch
                           pt_tokens TYPE /auditar/ia_tyt_tokens
                           ps_campos LIKE /auditar/iafl
                           ps_datos  TYPE /auditar/ia_tys_datos_cbte
                           value(pv_valor_orig)
                  CHANGING pv_valor
                           pv_ignore
                           pt_msg  TYPE /auditar/ia_tyt_msg_campo.

  TYPES: BEGIN OF tys_moneda_soc,
           bukrs TYPE t001-bukrs,
           waers TYPE t001-waers,
         END OF tys_moneda_soc,
         tyt_moneda_soc TYPE STANDARD TABLE OF tys_moneda_soc.

  STATICS: lt_moneda_soc TYPE tyt_moneda_soc.

  DATA: ls_moneda_soc TYPE tys_moneda_soc,
        lv_tabix      TYPE sy-tabix.

  FIELD-SYMBOLS: <fs_moneda_soc> TYPE tys_moneda_soc.

  READ TABLE lt_moneda_soc WITH KEY bukrs = ps_datos-sociedad
                           ASSIGNING <fs_moneda_soc>
                           BINARY SEARCH.
  IF sy-subrc <> 0.
    lv_tabix = sy-tabix.
    ASSIGN ls_moneda_soc TO <fs_moneda_soc>.
    <fs_moneda_soc>-bukrs = ps_datos-sociedad.
    SELECT SINGLE waers FROM t001
                        INTO <fs_moneda_soc>-waers
                        WHERE bukrs = ps_datos-sociedad.
    INSERT <fs_moneda_soc> INTO lt_moneda_soc INDEX lv_tabix.
  ENDIF.

  pv_valor = <fs_moneda_soc>-waers.

ENDFORM.                    "get_moneda_gral

*&---------------------------------------------------------------------*
*&      Form  get_nrotarj_from_equi_gral
*&---------------------------------------------------------------------*
FORM get_nrotarj_from_equi_gral USING pv_soc
                                      pv_parch
                                      pt_tokens  TYPE /auditar/ia_tyt_tokens
                                      ps_campos  LIKE /auditar/iafl
                                      ps_datos   TYPE /auditar/ia_tys_datos_cbte
                                      value(pv_valor_orig)
                             CHANGING pv_valor
                                      ps_param_out TYPE /auditar/ia_tys_param_out
                                      pt_msg       TYPE /auditar/ia_tyt_msg_campo.

  DATA: ls_msg        TYPE /auditar/ia_tys_msg_campo,
        lv_tarmov     LIKE /auditar/eqtjcb-tarmov,
        ls_equitarmov TYPE /auditar/eqtjcb,
        lv_tabix      LIKE sy-tabix,
        lv_no_existe.

  STATICS lt_equitarmov TYPE TABLE OF /auditar/eqtjcb.

  FIELD-SYMBOLS: <fs_equitarmov> TYPE /auditar/eqtjcb.

  lv_tarmov = pv_valor_orig.

  SHIFT lv_tarmov LEFT DELETING LEADING space.
  TRANSLATE lv_tarmov TO UPPER CASE.                     "#EC TRANSLANG

  READ TABLE lt_equitarmov WITH KEY sociedad = ps_datos-sociedad
                                    tarmov   = lv_tarmov
                                    tarjdeb  = ps_datos-tarjdebito
                           ASSIGNING <fs_equitarmov>
                           BINARY SEARCH.

  IF <fs_equitarmov> IS ASSIGNED.
    IF NOT <fs_equitarmov>-nrotarj IS INITIAL.
      pv_valor = <fs_equitarmov>-nrotarj.
    ELSE.
      lv_no_existe = 'X'.
    ENDIF.
  ELSE.
    lv_tabix = sy-tabix.
    SELECT SINGLE sociedad tarmov tarjdeb nrotarj
                  FROM /auditar/eqtjcb
                  INTO CORRESPONDING FIELDS OF ls_equitarmov
                  WHERE sociedad = ps_datos-sociedad
                    AND tarmov   = lv_tarmov
                    AND tarjdeb  = ps_datos-tarjdebito.
    IF sy-subrc = 0.
      ls_equitarmov-sociedad = ps_datos-sociedad.
      ls_equitarmov-tarmov   = lv_tarmov.
      ls_equitarmov-tarjdeb  = ps_datos-tarjdebito.
      pv_valor = ls_equitarmov-nrotarj.
    ELSE.
      lv_no_existe = 'X'.
    ENDIF.
    INSERT ls_equitarmov INTO lt_equitarmov INDEX lv_tabix.
  ENDIF.

  IF lv_no_existe = 'X'.
    PERFORM armar_y_agregar_msg  USING 'E'
                                       '/AUDITAR/01'
                                       '310'
                                       lv_tarmov
                                       space
                                       space
                                       space
                                 CHANGING pt_msg.
  ENDIF.

ENDFORM.                    "get_nrotarj_from_equi_gral


*&---------------------------------------------------------------------*
*&      Form  get_nrotarj_ztarmov_gral
*&---------------------------------------------------------------------*
FORM get_nrotarj_ztarmov_gral USING pv_soc
                                    pv_parch
                                    pt_tokens  TYPE /auditar/ia_tyt_tokens
                                    ps_campos  LIKE /auditar/iafl
                                    ps_datos   TYPE /auditar/ia_tys_datos_cbte
                                    value(pv_valor_orig)
                          CHANGING  pv_valor
                                    ps_param_out TYPE /auditar/ia_tys_param_out
                                    pt_msg       TYPE /auditar/ia_tyt_msg_campo.

  STATICS: lt_equitarmov       TYPE STANDARD TABLE OF /auditar/eqtjcb,
           lv_equitarmov_leida.

  DATA: lv_tabix         TYPE sytabix,
        lv_cant          TYPE i,
        lv_nrotarj       TYPE /auditar/nrotarj,
        lv_clave         LIKE /auditar/eqtjcb-tarmov,
        lv_mediopago     LIKE /auditar/eqtjcb-tarmov,
        lv_tarmov        LIKE /auditar/eqtjcb-tarmov,
        lv_interf_dummy  TYPE string,
        lv_nrotarj_dummy TYPE string,
        lv_codret.

  FIELD-SYMBOLS <fs_equitarmov> TYPE /auditar/eqtjcb.

  PERFORM get_compadic_char USING    ps_datos-adic
                                     'ZTARMOV'
                                     '0'
                            CHANGING lv_tarmov
                                     lv_codret
                                     pt_msg.
  SPLIT lv_tarmov AT '|' INTO lv_clave
                              lv_mediopago.

  IF lv_equitarmov_leida IS INITIAL.
    CONCATENATE lv_clave '%' INTO lv_clave.

    SELECT sociedad tarmov tarjdeb nrotarj              "#EC CI_GENBUFF
           FROM /auditar/eqtjcb
           INTO CORRESPONDING FIELDS OF TABLE lt_equitarmov
           WHERE tarmov LIKE lv_clave.

    LOOP AT lt_equitarmov ASSIGNING <fs_equitarmov>.
      SPLIT <fs_equitarmov>-tarmov AT '|' INTO lv_interf_dummy
                                               <fs_equitarmov>-tarmov
                                               lv_nrotarj_dummy.
    ENDLOOP.

    SORT lt_equitarmov BY sociedad tarmov tarjdeb nrotarj.
    lv_equitarmov_leida = 'X'.
  ENDIF.

  READ TABLE lt_equitarmov WITH KEY sociedad = ps_datos-sociedad
                                    tarmov   = lv_mediopago
                                    tarjdeb  = ps_datos-tarjdebito
                           BINARY SEARCH
                           TRANSPORTING NO FIELDS.

  IF sy-subrc = 0.
    lv_tabix = sy-tabix.
    LOOP AT lt_equitarmov FROM lv_tabix ASSIGNING <fs_equitarmov>.
      IF  <fs_equitarmov>-sociedad <> ps_datos-sociedad OR
          <fs_equitarmov>-tarmov   <> lv_mediopago      OR
          <fs_equitarmov>-tarjdeb  <> ps_datos-tarjdebito.
        EXIT.
      ENDIF.

      ADD 1 TO lv_cant.
      lv_nrotarj = <fs_equitarmov>-nrotarj.

      IF lv_cant > 1.
        EXIT.
      ENDIF.
    ENDLOOP.
  ENDIF.

  CASE lv_cant.
    WHEN 0.
      PERFORM armar_y_agregar_msg  USING 'E'
                                         '/AUDITAR/01'
                                         '310'
                                         lv_tarmov
                                         space
                                         space
                                         space
                                   CHANGING pt_msg.
    WHEN 1.
      pv_valor = lv_nrotarj.
    WHEN OTHERS.
      PERFORM armar_y_agregar_msg  USING 'E'
                                         '/AUDITAR/01'
                                         '311'
                                         lv_tarmov
                                         space
                                         space
                                         space
                                   CHANGING pt_msg.
  ENDCASE.

ENDFORM.        "get_nrotarj_ztarmov_gral

*&      Form  convertir_fecha_ddmmaa_gral
*&---------------------------------------------------------------------*
FORM convertir_fecha_ddmmaa_gral USING pv_soc
                                       pv_parch
                                       pt_tokens  TYPE /auditar/ia_tyt_tokens
                                       ps_campos  LIKE /auditar/iafl
                                       ps_datos   TYPE
                                       /auditar/ia_tys_datos_cbte
                                       value(pv_valor_orig)
                              CHANGING pv_valor
                                       ps_param_out TYPE /auditar/ia_tys_param_out
                                       pt_msg       TYPE /auditar/ia_tyt_msg_campo.

  DATA: lv_len       TYPE i,
        lv_anio(2)   TYPE n,
        lv_anio_f(4) TYPE n,
        ls_msg       LIKE LINE OF pt_msg.

  lv_len = strlen( pv_valor_orig ).
  IF lv_len <> 6 OR NOT pv_valor_orig(6) CO c_numbers.
    ls_msg-tipo  = 'E'.
    MESSAGE e353 INTO ls_msg-texto.
    APPEND ls_msg TO pt_msg.
    EXIT.
  ENDIF.

  lv_anio = pv_valor_orig+4(2).

  CALL FUNCTION 'CONVERT_YEAR_WITH_THRESHOLD'
    EXPORTING
      input  = lv_anio
    IMPORTING
      output = lv_anio_f.

  CONCATENATE lv_anio_f pv_valor_orig+2(2) pv_valor_orig(2) INTO
  pv_valor.

ENDFORM.                    "convertir_fecha_ddmma_gral


*&---------------------------------------------------------------------*
*&      Form  convertir_fecha_separador_gral
*&---------------------------------------------------------------------*
FORM convertir_fecha_separador_gral USING pv_soc
                                          pv_parch
                                          pt_tokens  TYPE /auditar/ia_tyt_tokens
                                          ps_campos  LIKE /auditar/iafl
                                          ps_datos   TYPE /auditar/ia_tys_datos_cbte
                                          value(pv_valor_orig)
                                 CHANGING pv_valor
                                          ps_param_out TYPE /auditar/ia_tys_param_out
                                          pt_msg       TYPE /auditar/ia_tyt_msg_campo.

  DATA: ls_msg      TYPE /auditar/ia_tys_msg_campo,
        lv_len      TYPE i,
        lv_fecha(8),

    lv_error = 'X'.
  ELSE.
    CONCATENATE pv_valor_orig+6(4)
                pv_valor_orig+3(2)
                pv_valor_orig(2) INTO lv_fecha.
    IF lv_fecha CN c_numbers.

      pv_valor = lv_fecha.
    ENDIF.
  ENDIF.

  IF lv_error = 'X'.

    APPEND ls_msg TO pt_msg.

*&---------------------------------------------------------------------*
*&      Form  CONVERT_FECHA_DMAAAA_SEPAR
*----------------------------------------------------------------------*
FORM convert_fecha_dmaaaa_separ USING pv_soc
                                      pv_parch
                                      pt_tokens  TYPE /auditar/ia_tyt_tokens
                                      ps_campos  LIKE /auditar/iafl
                                      ps_datos   TYPE /auditar/ia_tys_datos_cbte
                                      value(pv_valor_orig)
                             CHANGING pv_valor
                                      ps_param_out TYPE /auditar/ia_tys_param_out
                                      pt_msg       TYPE /auditar/ia_tyt_msg_campo.
  DATA: lv_dia(2)  TYPE n,
        lv_mes(2)  TYPE n,
        lv_anio(4) TYPE n.
  IF pv_valor_orig CS '/'.
    SPLIT pv_valor_orig AT '/' INTO lv_dia lv_mes lv_anio.
  ENDIF.
  IF pv_valor_orig CS '-'.
    SPLIT pv_valor_orig AT '-' INTO lv_dia lv_mes lv_anio.
  ENDIF.
  IF pv_valor_orig CS '.'.
    SPLIT pv_valor_orig AT '.' INTO lv_dia lv_mes lv_anio.
  ENDIF.
  SHIFT lv_dia RIGHT DELETING TRAILING space.
  OVERLAY lv_dia WITH '00'.
  SHIFT lv_mes RIGHT DELETING TRAILING space.
  OVERLAY lv_mes WITH '00'.
  CONCATENATE lv_anio lv_mes lv_dia INTO pv_valor.

ENDFORM.                    "CONVERT_FECHA_DMAAAA_SEPAR


*&---------------------------------------------------------------------*
*&      Form  get_nroestab_gral
*&---------------------------------------------------------------------*
FORM get_nroestab_gral USING pv_soc
                              pv_parch
                              pt_tokens  TYPE /auditar/ia_tyt_tokens
                              ps_campos  LIKE /auditar/iafl
                              ps_datos   TYPE /auditar/ia_tys_datos_cbte
                              value(pv_valor_orig)
                     CHANGING pv_valor
                              ps_param_out TYPE /auditar/ia_tys_param_out
                              pt_msg       TYPE /auditar/ia_tyt_msg_campo.


  STATICS: lt_estab        TYPE TABLE OF /auditar/es,
           lv_estab_leidos.

  DATA: lv_nroestab    LIKE /auditar/es-nroestab,
        lv_codextestab LIKE /auditar/es-codextestab,
        lv_tabix       TYPE sytabix,
        lv_cant        TYPE i.

  FIELD-SYMBOLS: <fs_estab> TYPE /auditar/es.

  IF lv_estab_leidos IS INITIAL.
    SELECT sociedad nrotarj codextestab nroestab        "#EC CI_GENBUFF
           FROM /auditar/es
           INTO CORRESPONDING FIELDS OF TABLE lt_estab.
    SORT lt_estab BY sociedad nrotarj codextestab.
    lv_estab_leidos = 'X'.
  ENDIF.

  CLEAR: pv_valor.

  SHIFT pv_valor_orig LEFT DELETING LEADING '0'.
  CONDENSE pv_valor_orig.
  TRANSLATE pv_valor_orig TO UPPER CASE.
  lv_codextestab = pv_valor_orig.

  " Obtengo nro estab segun Auditar
  READ TABLE lt_estab WITH KEY sociedad    = ps_datos-sociedad
                               nrotarj     = ps_datos-nrotarj
                               codextestab = lv_codextestab
                      TRANSPORTING NO FIELDS
                      BINARY SEARCH.
  IF sy-subrc = 0.
    lv_tabix = sy-tabix.
    LOOP AT lt_estab ASSIGNING <fs_estab> FROM lv_tabix.
      IF <fs_estab>-codextestab <> lv_codextestab OR
         <fs_estab>-nrotarj <> ps_datos-nrotarj OR
         <fs_estab>-sociedad <> ps_datos-sociedad.
        EXIT.
      ENDIF.
      lv_nroestab = <fs_estab>-nroestab.
      ADD 1 TO lv_cant.
    ENDLOOP.
  ENDIF.


  CASE lv_cant.
    WHEN 0.
      PERFORM armar_y_agregar_msg  USING    'E'
                                            '/AUDITAR/01'
                                            '314'
                                            ps_datos-sociedad
                                            ps_datos-nrotarj
                                            lv_codextestab
                                            space
                                   CHANGING pt_msg.
    WHEN 1.
      pv_valor = lv_nroestab.
    WHEN OTHERS.
      PERFORM armar_y_agregar_msg  USING    'E'
                                            '/AUDITAR/01'
                                            '312'
                                            ps_datos-sociedad
                                            ps_datos-nrotarj
                                            lv_codextestab
                                            space
                                   CHANGING pt_msg.

  ENDCASE.

ENDFORM.                    "get_nroestab_gral

                            ps_datos   TYPE /auditar/ia_tys_datos_cbte

                   CHANGING pv_valor
                            ps_param_out TYPE /auditar/ia_tys_param_out
                            pt_msg       TYPE /auditar/ia_tyt_msg_campo.

  DATA: lv_feclanzam TYPE /auditar/mig-fechalanzam,
        lv_fecvta    LIKE /auditar/cb-feccbte,
        lv_soc       LIKE /auditar/saldos-sociedad,
        lv_nrotarj   LIKE /auditar/saldos-nrotarj,

* Obtengo la fecha de venta segun interfase
  lv_soc     = ps_datos-sociedad.

  ls_cbte-sociedad = lv_soc.

  PERFORM leer_grupxestab USING ps_datos-sociedad

    READ TABLE lt_parvarios INDEX 1 ASSIGNING <fs_parvarios>.

  FIELD-SYMBOLS: <fs_grupxestab> TYPE tys_grupxestab,
                 <fs_parvarios>  TYPE /auditar/dparv.

  CLEAR lv_marca.

  IF pv_valor_orig CO ` 0`.
    lv_cuotas = '1'.
  ELSE.
    lv_cuotas = pv_valor_orig.
  ENDIF.

  " Busco el parámetro que identifica grupo utilizado
  CALL FUNCTION '/AUDITAR/GET_PARVARIOS'
    EXPORTING
      i_sociedad    = ps_datos-sociedad
      i_proceso     = gc_proceso_cont
      i_parametro   = 'GRPES'
      i_valaper     = 'ACEPTA_PLAN_GOBIERNO'
    IMPORTING
      et_dparvarios = lt_parvarios[].

  " si no existe doy error
  IF  lt_parvarios[] IS INITIAL.
    ls_msg-tipo = 'E'.
    MESSAGE e464 WITH 'GRPES'
                      'ACEPTA_PLAN_GOBIERNO'
                 INTO ls_msg-texto.
    APPEND ls_msg TO pt_msg.
    EXIT.
  ENDIF.

  READ TABLE lt_parvarios ASSIGNING <fs_parvarios> INDEX 1.
  lv_grupo = <fs_parvarios>-valor.

  " Busco el dato cargado en el grupo para el establecimiento
  PERFORM leer_grupxestab USING ps_datos-sociedad
                                ps_datos-nrotarj
                                ps_datos-nroestab
                                lv_grupo
                       CHANGING lt_grupxestab.

  "si no existe no se considera el grupo para la búsqueda
  IF lt_grupxestab[] IS INITIAL.
    CONCATENATE ps_datos-nrotarj lv_cuotas INTO lv_valaper SEPARATED BY '-'.
  ELSE.
    READ TABLE lt_grupxestab ASSIGNING <fs_grupxestab> INDEX 1.
    CONCATENATE ps_datos-nrotarj <fs_grupxestab>-valor lv_cuotas INTO lv_valaper SEPARATED BY '-'.
  ENDIF.

  CALL FUNCTION '/AUDITAR/GET_PARVARIOS'
    EXPORTING
      i_sociedad    = ps_datos-sociedad
      i_proceso     = gc_proceso_cbtes
      i_parametro   = 'CBGOB'
      i_valaper     = lv_valaper
    IMPORTING
      et_dparvarios = lt_parvarios.

  IF NOT lt_parvarios[] IS INITIAL.
    READ TABLE lt_parvarios INDEX 1 ASSIGNING <fs_parvarios>.
    SPLIT <fs_parvarios>-valor AT '-' INTO lv_marca lv_cuotas.
  ENDIF.

  pv_valor = lv_cuotas.

ENDFORM.                    "get_cuotas_gral


*&---------------------------------------------------------------------*
*&      Form  leer_grupxestab
*&---------------------------------------------------------------------*
FORM leer_grupxestab USING pv_sociedad
                           pv_nrotarj
                           pv_nroestab
                           pv_codgrupo
                  CHANGING pt_grupxestab TYPE tyt_grupxestab.

  DATA: lv_index           LIKE sy-tabix,
        ls_grupxestab      TYPE tys_grupxestab,
        ls_grupxestab_buff TYPE tys_grupxestab_buff.

  FIELD-SYMBOLS: <fs_grupxestab>      TYPE tys_grupxestab,
                 <fs_grupxestab_buff> TYPE tys_grupxestab_buff.

  STATICS: lt_grupxestab_buff TYPE tyt_grupxestab_buff.

  READ TABLE lt_grupxestab_buff WITH KEY sociedad = pv_sociedad
                                         nrotarj  = pv_nrotarj
                                         nroestab = pv_nroestab
                                         codgrupo = pv_codgrupo
                                TRANSPORTING NO FIELDS
                                BINARY SEARCH.
  lv_index = sy-tabix.
  IF sy-subrc = 0.
    LOOP AT lt_grupxestab_buff ASSIGNING <fs_grupxestab_buff>
                               FROM lv_index.

      IF <fs_grupxestab_buff>-sociedad <> pv_sociedad OR
         <fs_grupxestab_buff>-nrotarj  <> pv_nrotarj  OR
         <fs_grupxestab_buff>-nroestab <> pv_nroestab OR
         <fs_grupxestab_buff>-codgrupo <> pv_codgrupo.
        EXIT.
      ENDIF.

      IF <fs_grupxestab_buff>-no_existe = 'X'.
        EXIT.
      ENDIF.

      ls_grupxestab-valor = <fs_grupxestab_buff>-valor.

      APPEND ls_grupxestab TO pt_grupxestab.
    ENDLOOP.

  ELSE.
    ls_grupxestab_buff-sociedad  = pv_sociedad.
    ls_grupxestab_buff-nrotarj   = pv_nrotarj.
    ls_grupxestab_buff-nroestab  = pv_nroestab.
    ls_grupxestab_buff-codgrupo  = pv_codgrupo.

    SELECT valor FROM /auditar/grxes INTO TABLE pt_grupxestab
                 WHERE sociedad = pv_sociedad
                   AND nrotarj  = pv_nrotarj
                   AND nroestab = pv_nroestab
                   AND codgrupo = pv_codgrupo.
    IF sy-subrc = 0.
      SORT pt_grupxestab BY valor.

      LOOP AT pt_grupxestab ASSIGNING <fs_grupxestab>.
        ls_grupxestab_buff-valor = <fs_grupxestab>-valor.
        INSERT ls_grupxestab_buff INTO  lt_grupxestab_buff
                                  INDEX lv_index.
        ADD 1 TO lv_index.
      ENDLOOP.
    ELSE.
      ls_grupxestab_buff-no_existe = 'X'.
      INSERT ls_grupxestab_buff INTO  lt_grupxestab_buff
                                INDEX lv_index.
    ENDIF.
  ENDIF.
ENDFORM.                    "leer_grupxestab


*&---------------------------------------------------------------------*
*& Form APLANAR_ESTRUCTURA
*&---------------------------------------------------------------------*
FORM aplanar_estructura USING ps_data TYPE any
                              pv_deep_level
                              pv_field_sep
                     CHANGING value(pv_level)
                              pv_registro TYPE string
                              pt_archivo  TYPE tyt_archivo.

  DATA: lr_desc TYPE REF TO cl_abap_typedescr.
  DATA: lr_sdes TYPE REF TO cl_abap_structdescr.
  DATA: ls_comp TYPE cl_abap_structdescr=>component.
  DATA: lt_comp   TYPE cl_abap_structdescr=>component_table,
        lv_string TYPE string.
  FIELD-SYMBOLS: <fs_field> TYPE any.


  lr_desc = cl_abap_typedescr=>describe_by_data( ps_data ).

  CASE lr_desc->type_kind.

    WHEN cl_abap_typedescr=>typekind_struct1 OR cl_abap_typedescr=>typekind_struct2.
*     Para las esructuras, recorrer los campos y llamar al método de nuevo
      lr_sdes ?= lr_desc.
      lt_comp = lr_sdes->get_components( ).
      LOOP AT lt_comp INTO ls_comp.
        ASSIGN COMPONENT ls_comp-name OF STRUCTURE ps_data TO <fs_field>.
        IF sy-subrc = 0.
          PERFORM aplanar_estructura USING    <fs_field>
                                              pv_deep_level
                                              pv_field_sep
                                     CHANGING pv_level
                                              pv_registro
                                              pt_archivo.
        ENDIF.
      ENDLOOP.

    WHEN cl_abap_typedescr=>typekind_table.
*     Para las tablas internas, recorrer los registros y llamar al método de nuevo
      DATA: lv_line TYPE REF TO data.
      FIELD-SYMBOLS: <fs_tab> TYPE ANY TABLE.
      FIELD-SYMBOLS: <fs_line> TYPE any.

      ASSIGN ps_data TO <fs_tab>.
      CREATE DATA lv_line LIKE LINE OF <fs_tab>.
      ASSIGN lv_line->* TO <fs_line>.

      ADD 1 TO pv_level.

      CHECK pv_level <= pv_deep_level.

      LOOP AT <fs_tab> INTO <fs_line>.
        PERFORM aplanar_estructura USING    <fs_line>
                                            pv_deep_level
                                            pv_field_sep
                                   CHANGING pv_level
                                            pv_registro
                                            pt_archivo.
        APPEND pv_registro TO pt_archivo.
        CLEAR pv_registro.
      ENDLOOP.


    WHEN cl_abap_typedescr=>typekind_dref.
*     Para las referencias de datos, desreferenciar los datos y llamar al método de nuevo
      FIELD-SYMBOLS: <fs_data> TYPE any.

      ASSIGN ps_data->* TO <fs_data>.
      IF sy-subrc = 0.
        PERFORM aplanar_estructura USING    <fs_data>
                                            pv_deep_level
                                            pv_field_sep
                                   CHANGING pv_level
                                            pv_registro
                                            pt_archivo.
      ENDIF.

    WHEN OTHERS.
*     Para los campos, ir concatenando uno a uno los valores con el separador
      IF pv_registro IS INITIAL.
        pv_registro = ps_data.
      ELSE.
*        if cl_abap_typedescr=>TYPEKIND_PACKED.
*        else.
*        endif.
        lv_string = ps_data.
        CONCATENATE pv_registro lv_string INTO pv_registro SEPARATED BY pv_field_sep.
      ENDIF.
  ENDCASE.

ENDFORM.                    "aplanar_estructura
